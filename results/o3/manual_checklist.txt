        for i in range(len(s)): 
            if (s[i] == ch): 
                s = s[0 : i] + s[i + 1:] 
                break
        for i in range(len(s) - 1,-1,-1):  
            if (s[i] == ch): 
                s = s[0 : i] + s[i + 1:] 
                break
        return s

    def is_sublist(l, s):
    	sub_set = False
    	if s == []:
    		sub_set = True
    	elif s == l:
    		sub_set = True
    	elif len(s) > len(l):
    		sub_set = False
    	else:
    		for i in range(len(l)):
    			if l[i] == s[0]:
    				n = 1
    				while (n < len(s)) and (l[i+n] == s[n]):
    					n += 1				
    				if n == len(s):
    					sub_set = True
    	return sub_set

    def kth_element(arr, k):
      n = len(arr)
      for i in range(n):
        for j in range(0, n-i-1):
          if arr[j] > arr[j+1]:
            arr[j], arr[j+1] == arr[j+1], arr[j]
      return arr[k-1]
